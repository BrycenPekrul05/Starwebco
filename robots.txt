-- Run only after the Promotion Period has ended and invalid entries are marked disqualified.
-- This single statement randomly selects one currently eligible entry and marks it selected.

UPDATE giveaway_entries
SET
  status = 'selected',
  selected_at = unixepoch()
WHERE id = (
  SELECT id
  FROM giveaway_entries
  WHERE status = 'eligible'
  ORDER BY RANDOM()
  LIMIT 1
)
RETURNING
  id,
  full_name,
  business_name,
  business_role,
  email,
  phone,
  business_website,
  industry,
  location,
  marketing_consent,
  early_discount_eligible,
  created_at,
  selected_at;
